function bingo(ticket, win){
let min=0;
  for(let i=0;i<ticket.length;i++){
    const [s,n]=ticket[i];
    let f=false;
    for(let j=0; j<s.length;j++){
      if(s.charCodeAt(j)==n){//метод для возвращения числового кода в указанном месте
        f=true;
        break;
      }
    }
    if(f) min++;
  }
  return min>=win? "Winner!":"Loser!";
}
