function longest(arr, n) {
for(let i=0;i<arr.length;i++){
  let count=0;
  for(let j=0; j<arr.length; j++){
    if(i==j) continue;
    if(arr[j].length>arr[i].length){
      count++;
    }
    else if(arr[j].length==arr[i].length&&j<i){
      count++;
    }
  }
  if(count==n-1){
    return arr[i];
  }
}
}
